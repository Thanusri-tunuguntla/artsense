        // Show/hide sections
        function showSection(sectionId) {
            const sections = document.querySelectorAll('.section');
            const navBtns = document.querySelectorAll('.nav-btn');
            
            sections.forEach(s => s.classList.remove('active'));
            navBtns.forEach(btn => btn.classList.remove('active'));

            event.target.classList.add('active');

                displayProducts();

        // Filter products
        function filterProducts() {
            const category = document.getElementById('categoryFilter').value;
            const maxPrice = parseFloat(document.getElementById('priceFilter').value) || Infinity;
            const search = document.getElementById('searchFilter').value.toLowerCase();
            
            const filtered = products.filter(product => {
                const matchesCategory = !category || product.category === category;
                const matchesPrice = product.price <= maxPrice;
                const matchesSearch = !search || 
                    product.name.toLowerCase().includes(search) ||
                    product.description.toLowerCase().includes(search);
                
                return matchesCategory && matchesPrice && matchesSearch;
            });
            
            displayProducts(filtered);
        }

        // Contact seller
        function contactSeller(productId) {
            const product = products.find(p => p.id === productId);
            if (product) {
                const message = `Hi ${product.seller.name}, I'm interested in your ${product.name} listed for ‚Çπ${product.price.toLocaleString('en-IN')}. Can we discuss?`;
                const whatsappUrl = `https://wa.me/${product.seller.phone.replace(/\D/g, '')}?text=${encodeURIComponent(message)}`;
                window.open(whatsappUrl, '_blank');
            }
        }

        function handleImageUpload(file) {
            if (file.type.startsWith('image/')) {
                const reader = new FileReader();
                reader.onload = (e) => {
                    imagePreview.src = e.target.result;
                    imagePreview.style.display = 'block';
                    generateAIDescription(file);
                };
                reader.readAsDataURL(file);
            }
        }
